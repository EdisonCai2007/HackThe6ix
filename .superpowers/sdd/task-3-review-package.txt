     1	const DEFAULT_BASE_URL = "https://openrouter.ai/api/v1/chat/completions";
     2	
     3	export function createOpenRouterClient({
     4	  apiKey,
     5	  fetchImpl = fetch,
     6	  baseUrl = DEFAULT_BASE_URL,
     7	} = {}) {
     8	  if (!apiKey) {
     9	    throw new Error("OPENROUTER_API_KEY is required.");
    10	  }
    11	
    12	  return {
    13	    async complete(request) {
    14	      const response = await fetchImpl(baseUrl, {
    15	        method: "POST",
    16	        headers: {
    17	          Authorization: `Bearer ${apiKey}`,
    18	          "Content-Type": "application/json",
    19	          "HTTP-Referer": "http://127.0.0.1:5173",
    20	          "X-Title": "HackThe6ix LEGO Generator",
    21	        },
    22	        body: JSON.stringify(request),
    23	      });
    24	
    25	      const body = await response.json().catch(() => null);
    26	
    27	      if (!response.ok) {
    28	        const message = body?.error?.message ?? `OpenRouter request failed with ${response.status}`;
    29	        throw new Error(message);
    30	      }
    31	
    32	      const content = body?.choices?.[0]?.message?.content;
    33	
    34	      if (typeof content !== "string" || content.trim() === "") {
    35	        throw new Error("OpenRouter response did not include message content.");
    36	      }
    37	
    38	      return content;
    39	    },
    40	  };
    41	}

     1	import { parseJsonObject, parseStructurePlanText } from "./designPlan.js";
     2	import { validateGeneratedModelShape } from "./generatedModelSchema.js";
     3	import { buildPlacementPrompt, buildStructurePrompt } from "./openRouterPrompts.js";
     4	import { validateModel } from "./validator.js";
     5	
     6	function failure(stage, errors, extra = {}) {
     7	  return {
     8	    ok: false,
     9	    stage,
    10	    errors,
    11	    ...extra,
    12	  };
    13	}
    14	
    15	export async function generateModel({
    16	  userPrompt,
    17	  inventory,
    18	  targetPieceCount,
    19	  openRouterClient,
    20	  model,
    21	}) {
    22	  const structureRequest = buildStructurePrompt({
    23	    userPrompt,
    24	    inventory,
    25	    targetPieceCount,
    26	    model,
    27	  });
    28	
    29	  const structureText = await openRouterClient.complete(structureRequest);
    30	  const structureResult = parseStructurePlanText(structureText);
    31	
    32	  if (!structureResult.ok) {
    33	    return failure("structure_parse", structureResult.errors);
    34	  }
    35	
    36	  const placementRequest = buildPlacementPrompt({
    37	    userPrompt,
    38	    inventory,
    39	    structurePlan: structureResult.value,
    40	    targetPieceCount,
    41	    model,
    42	  });
    43	
    44	  const placementText = await openRouterClient.complete(placementRequest);
    45	  const placementJson = parseJsonObject(placementText, "placement model");
    46	
    47	  if (!placementJson.ok) {
    48	    return failure("placement_parse", placementJson.errors, {
    49	      structurePlan: structureResult.value,
    50	    });
    51	  }
    52	
    53	  const shapeResult = validateGeneratedModelShape(placementJson.value);
    54	
    55	  if (!shapeResult.ok) {
    56	    return failure("placement_shape", shapeResult.errors, {
    57	      structurePlan: structureResult.value,
    58	    });
    59	  }
    60	
    61	  const validation = validateModel(placementJson.value, inventory);
    62	
    63	  if (!validation.valid) {
    64	    return failure("validation", validation.errors, {
    65	      structurePlan: structureResult.value,
    66	      model: placementJson.value,
    67	      validation,
    68	    });
    69	  }
    70	
    71	  return {
    72	    ok: true,
    73	    stage: "complete",
    74	    structurePlan: structureResult.value,
    75	    model: placementJson.value,
    76	    validation,
    77	  };
    78	}
