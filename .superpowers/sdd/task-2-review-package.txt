     5	function clampTargetPieceCount(targetPieceCount) {
     6	  if (!Number.isFinite(targetPieceCount)) {
     7	    return Math.min(40, MAX_MODEL_PIECES);
     8	  }
     9	
    10	  return Math.max(1, Math.min(Math.floor(targetPieceCount), MAX_MODEL_PIECES));
    11	}

    13	export function summarizeSupportedInventory(inventory) {
    14	  return {
    15	    inventory_id: inventory.inventory_id,
    16	    source: inventory.source,
    17	    items: inventory.items
    18	      .filter((item) => item.supported && SUPPORTED_PARTS[item.part_id])
    19	      .map((item) => ({
    20	        label: item.label,
    21	        category: item.category,
    22	        part_id: item.part_id,
    23	        ldraw_id: item.ldraw_id,
    24	        color_name: item.color_name,
    25	        color_id: item.color_id,
    26	        count: item.count,
    27	        supported: true,
    28	        dimensions: {
    29	          width: SUPPORTED_PARTS[item.part_id].width,
    30	          depth: SUPPORTED_PARTS[item.part_id].depth,
    31	          height_layers: SUPPORTED_PARTS[item.part_id].category === "plate" ? 1 : 3,
    32	        },
    33	      })),
    34	  };
    35	}
