    47	export function validateStructurePlan(plan) {
    48	  const errors = [];
    49	
    50	  if (!isPlainObject(plan)) {
    51	    return {
    52	      ok: false,
    53	      errors: [issue("structure_plan", "Structure plan must be an object.")],
    54	    };
    55	  }
    56	
    57	  for (const field of ["model_name", "primary_object", "overall_shape", "user_facing_summary"]) {
    58	    requireString(plan, field, errors);
    59	  }
    60	
    61	  requirePositiveNumber(plan, "target_piece_count", errors);
    62	  requireNonEmptyArray(plan, "required_features", errors);
    63	  requireNonEmptyArray(plan, "part_usage_plan", errors);
    64	  requireNonEmptyArray(plan, "fallback_priorities", errors);
    65	
    66	  if (!isPlainObject(plan.build_strategy)) {
    67	    errors.push(issue("build_strategy", "build_strategy must be an object."));
    68	  }
    69	
    70	  return {
    71	    ok: errors.length === 0,
    72	    errors,
    73	  };
    74	}

    35	  if (!isPlainObject(position)) {
    36	    errors.push(issue(`bricks[${index}].position`, "position must be an object."));
    37	    return;
    38	  }
    39	
    40	  for (const axis of ["x", "y", "z"]) {
    41	    requireNumber(position[axis], `bricks[${index}].position.${axis}`, errors);
    42	  }
    43	}

    45	function validateBrick(brick, index, errors) {
    46	  if (!isPlainObject(brick)) {
    47	    errors.push(issue(`bricks[${index}]`, "brick must be an object."));
    48	    return;
    49	  }
    50	
    51	  for (const field of [
    52	    "id",
    53	    "part_id",
    54	    "ldraw_id",
    55	    "label",
    56	    "color_id",
    57	    "color_name",
    58	    "feature",
    59	  ]) {
    60	    requireString(brick[field], `bricks[${index}].${field}`, errors);
    61	  }
    62	
    63	  validatePosition(brick.position, index, errors);
    64	
    65	  if (!VALID_ROTATIONS.has(brick.rotation)) {
    66	    errors.push(issue(`bricks[${index}].rotation`, "rotation must be 0, 90, 180, or 270."));
    67	  }
    68	
    69	  if (!Number.isInteger(brick.step) || brick.step <= 0) {
    70	    errors.push(issue(`bricks[${index}].step`, "step must be a positive integer."));
    71	  }
    72	}

    74	export function validateGeneratedModelShape(model) {
    75	  const errors = [];
    76	
    77	  if (!isPlainObject(model)) {
    78	    return {
    79	      ok: false,
    80	      errors: [issue("model", "Generated model must be an object.")],
    81	    };
    82	  }
    83	
    84	  for (const field of [
    85	    "model_name",
    86	    "prompt",
    87	    "created_from_inventory_id",
    88	    "generator_version",
    89	  ]) {
    90	    requireString(model[field], field, errors);
    91	  }
    92	
    93	  if (!Number.isInteger(model.piece_count) || model.piece_count < 0) {
    94	    errors.push(issue("piece_count", "piece_count must be a non-negative integer."));
    95	  }
    96	
    97	  validateDimensions(model.dimensions, errors);
    98	
    99	  if (!Array.isArray(model.bricks)) {
   100	    errors.push(issue("bricks", "bricks must be an array."));
   101	  } else {
   102	    model.bricks.forEach((brick, index) => validateBrick(brick, index, errors));
   103	  }
   104	
   105	  if (!Array.isArray(model.notes)) {
   106	    errors.push(issue("notes", "notes must be an array."));
   107	  }
   108	
   109	  return {
   110	    ok: errors.length === 0,
   111	    errors,
   112	  };
   113	}

   152	  }
   153	
   154	  const normalizedRotation = ((rotation % 360) + 360) % 360;
   155	
   156	  if (normalizedRotation === 90 || normalizedRotation === 270) {
   157	    return {
   158	      width: part.depth,
   159	      depth: part.width,
   160	      height: PART_HEIGHTS[part.category] ?? PART_HEIGHTS.brick,
   161	    };
   162	  }
   163	
   164	  return {
   165	    width: part.width,
   166	    depth: part.depth,
   167	    height: PART_HEIGHTS[part.category] ?? PART_HEIGHTS.brick,
   168	  };
