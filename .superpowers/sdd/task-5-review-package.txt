    94	function clearCurrentModel() {
    95	  if (!currentModelGroup) {
    96	    return;
    97	  }
    98	
    99	  scene.remove(currentModelGroup);
   100	  currentModelGroup.traverse((child) => {
   101	    child.geometry?.dispose?.();
   102	    if (Array.isArray(child.material)) {
   103	      child.material.forEach((material) => material.dispose?.());
   104	    } else {
   105	      child.material?.dispose?.();
   106	    }
   107	  });
   108	  currentModelGroup = null;
   109	}

   111	function renderModel(model) {
   112	  const loader = new LDrawLoader();
   113	  loader.setConditionalLineMaterial(LDrawConditionalLineMaterial);
   114	  const ldrawText = exportModelToLDraw(model);
   115	
   116	  loader.parse(
   117	    ldrawText,
   118	    (group) => {
   119	      clearCurrentModel();
   120	      group.rotation.x = Math.PI;
   121	      group.traverse((child) => {
   122	        if (child.isMesh) {
   123	          child.castShadow = true;
   124	          child.receiveShadow = true;
   125	        }
   126	      });
   127	
   128	      const box = new THREE.Box3().setFromObject(group);
   129	      const size = box.getSize(new THREE.Vector3());
   130	      const center = box.getCenter(new THREE.Vector3());
   131	      group.position.sub(center);
   132	
   133	      scene.add(group);
   134	      currentModelGroup = group;
   135	      controls.target.set(0, 0, 0);
   136	
   137	      const maxDimension = Math.max(size.x, size.y, size.z);
   138	      const fov = THREE.MathUtils.degToRad(camera.fov);
   139	      const distance = (maxDimension / (2 * Math.tan(fov / 2))) * 1.45;
   140	      const viewDirection = new THREE.Vector3(1.1, 0.75, 1.35).normalize();
   141	
   142	      camera.position.copy(viewDirection.multiplyScalar(distance));
   143	      camera.near = Math.max(0.1, distance / 100);
   144	      camera.far = distance * 10;
   145	      camera.updateProjectionMatrix();
   146	      controls.minDistance = distance * 0.35;
   147	      controls.maxDistance = distance * 2.2;
   148	      controls.update();
   149	    },
   150	    (error) => {
   151	      validationStatus.textContent = "Load error";
   152	      validationErrors.hidden = false;
   153	      validationErrors.textContent = error.message;
   154	    },
   155	  );
   156	}

   187	async function requestGeneration() {
   188	  const response = await fetch("http://127.0.0.1:8787/api/generate", {
   189	    method: "POST",
   190	    headers: { "Content-Type": "application/json" },
   191	    body: JSON.stringify({
   192	      userPrompt: promptInput.value,
   193	      inventory: selectedInventory(),
   194	      targetPieceCount: Number(targetPiecesInput.value),
   195	    }),
   196	  });
   197	
   198	  const result = await response.json();
   199	
   200	  if (!response.ok || !result.ok) {
   201	    throw result;
   202	  }
   203	
   204	  return result;
   205	}
