     6	function clampTargetPieceCount(targetPieceCount) {
     7	  if (!Number.isFinite(targetPieceCount)) {
     8	    return Math.min(40, MAX_MODEL_PIECES);
     9	  }
    10	
    11	  return Math.max(1, Math.min(Math.floor(targetPieceCount), MAX_MODEL_PIECES));
    12	}

    14	export function summarizeSupportedInventory(inventory) {
    15	  return {
    16	    inventory_id: inventory.inventory_id,
    17	    source: inventory.source,
    18	    items: inventory.items
    19	      .filter((item) => item.supported && SUPPORTED_PARTS[item.part_id])
    20	      .map((item) => ({
    21	        label: item.label,
    22	        category: item.category,
    23	        part_id: item.part_id,
    24	        ldraw_id: item.ldraw_id,
    25	        color_name: item.color_name,
    26	        color_id: item.color_id,
    27	        count: item.count,
    28	        supported: true,
    29	        dimensions: {
    30	          width: SUPPORTED_PARTS[item.part_id].width,
    31	          depth: SUPPORTED_PARTS[item.part_id].depth,
    32	          height_layers: SUPPORTED_PARTS[item.part_id].category === "plate" ? 1 : 3,
    33	        },
    34	      })),
    35	  };
    36	}
